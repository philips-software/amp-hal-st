add_library(hal_st.instantiations_lwip ${HALST_EXCLUDE_FROM_ALL} STATIC)

    set_property(TARGET hal_st.instantiations_lwip PROPERTY EXCLUDE_FROM_ALL TRUE)

    "$<BUILD_INTERFACE:${CMAKE_CURRENT_LIST_DIR}/../..>"

)

    infra.event
    infra.timer
    lwip.lwip_cpp

    hal_st.stm32fxxx
)
