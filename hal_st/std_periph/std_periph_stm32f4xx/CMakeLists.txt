    "$<BUILD_INTERFACE:${CMAKE_CURRENT_LIST_DIR}/inc>"
    "$<BUILD_INTERFACE:${CMAKE_CURRENT_LIST_DIR}/inc/cmsis>"
    "$<BUILD_INTERFACE:${CMAKE_CURRENT_LIST_DIR}/inc/stm32f4-hal>"
    "$<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}/inc>"
    "$<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}/inc/cmsis>"
    "$<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}/inc/stm32f4-hal>"
)

    hal_st.cmsis_arm
    hal_st.cmsis_cortex
)

target_compile_definitions(hal_st.std_periph_stm32f4xx PUBLIC
    USE_HAL_DRIVER
    HSE_VALUE=8000000
    STM32F4
    $<$<STREQUAL:${TARGET_MCU},stm32f407>:STM32F407xx>
    $<$<STREQUAL:${TARGET_MCU},stm32f429>:STM32F429xx>
    $<$<BOOL:${EMIL_HOST_BUILD}>:STM32F407xx> # For syntax recognition
    $<$<NOT:$<CONFIG:MinSizeRel>>:USE_FULL_ASSERT>
    $<$<NOT:$<CONFIG:MinSizeRel>>:TRACE>
    $<$<NOT:$<CONFIG:MinSizeRel>>:OS_USE_TRACE_ITM>
)
